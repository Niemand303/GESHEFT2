# Reunificate Albania and set politic
country_event = {
	id = border.1
	title = border.1.t
	desc = border.1.d
	picture = GFX_no
	hidden = yes
	fire_only_once = yes
	trigger = {
		country_exists = ALB
		OR = { # Possible circumstance once AXIS capitulate
			country_exists = WGR
			country_exists = DDR
			GER= {
				OR = {
					is_puppet = yes
					has_government = communism
					has_government = democratic
				}
			}
			ITA = {
				OR = {
					is_puppet = yes
					has_government = communism
					has_government = democratic
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 0
	}

	option ={
		if = {
			limit = { # YUG or ITA is com.
				OR = {
					YUG = { has_government = communism }
					ITA = { has_government = communism }
				}
			}
			ALB = {
				set_politics = {
					ruling_party = communism
					elections_allowed = no
					parties = {
						democratic = { 
							popularity = 0
						}
						fascism = {
							popularity = 0
						}
						communism = {
							popularity = 100
						}
					}
				}
			}
		}
		if = {
			limit = { # YUG is dem.
				YUG = { has_government = democratic }
			}
			ALB = {
				set_politics = {
					ruling_party = democratic
					elections_allowed = yes
					parties = {
						democratic = { 
							popularity = 100
						}
						fascism = {
							popularity = 0
						}
						communism = {
							popularity = 0
						}
					}
				}
			}
		}
		#Return ALB territory
		every_state = {
			limit = {
				is_core_of = ALB
				NOT = {
					is_owned_and_controlled_by = ALB
				}
			}
			ALB = {
				transfer_state = PREV
			}
		}
	}
}
